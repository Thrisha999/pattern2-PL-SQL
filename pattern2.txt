--PL/SQL Block;
declare
	n int;
	i int;
	j int;
begin
	n:=&rows;
	for i in reverse 1..n
	loop
	 for j in 1..i
	 loop
	    dbms_output.put('*');
	 end loop;
	 dbms_output.new_line();
	end loop;
end;
/
